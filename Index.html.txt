<script>
    // LISTA DE PROFESSORES (Edite aqui para adicionar ou remover nomes)
    const PROFESSORES = [
        "Adriano Silva",
        "Beatriz Souza",
        "Carlos Eduardo",
        "Daniela Oliveira",
        "Fernando Costa",
        "Gisele Martins",
        "Helena Rocha",
        "João Paulo",
        "Juliana Lima",
        "Marcos Vinícius",
        "Patrícia Gomes",
        "Ricardo Santos"
    ].sort(); // Ordena a lista de nomes para o select

    function popularSelect() {
        const select = document.getElementById('professorSelect');
        PROFESSORES.forEach(prof => {
            let opt = document.createElement('option');
            opt.value = prof;
            opt.innerHTML = prof;
            select.appendChild(opt);
        });
    }

    function verificarHorarioPermitido() {
        const agora = new Date();
        const horaAtual = agora.getHours();
        
        // Regra: Bloqueado entre 08:00 e 11:59:59
        if (horaAtual >= 8 && horaAtual < 12) {
            return {
                permitido: false,
                msg: "SISTEMA BLOQUEADO (Inscrições das 12h às 08h)"
            };
        }
        return { permitido: true, msg: "SISTEMA ABERTO" };
    }

    function atualizarInterface() {
        const status = verificarHorarioPermitido();
        const btn = document.getElementById('btnInscricao');
        const statusArea = document.getElementById('statusArea');

        if (!status.permitido) {
            btn.disabled = true;
            statusArea.textContent = status.msg;
            statusArea.className = "status-msg error";
        } else {
            btn.disabled = false;
            statusArea.textContent = status.msg;
            statusArea.className = "status-msg success";
        }
        renderizarListas();
    }

    function registrar() {
        const nome = document.getElementById('professorSelect').value;
        const horario = document.getElementById('horario').value;

        if (!nome) {
            alert("Por favor, selecione seu nome na lista.");
            return;
        }

        let inscritos = JSON.parse(localStorage.getItem('iern_almoco') || "[]");
        
        // Evita que o mesmo professor se inscreva duas vezes no mesmo ciclo
        if (inscritos.some(p => p.nome === nome)) {
            alert("Este professor já está inscrito na lista.");
            return;
        }

        inscritos.push({ nome, horario });
        localStorage.setItem('iern_almoco', JSON.stringify(inscritos));
        
        alert("Inscrição confirmada!");
        document.getElementById('professorSelect').value = "";
        renderizarListas();
    }

    function renderizarListas() {
        const inscritos = JSON.parse(localStorage.getItem('iern_almoco') || "[]");
        const lista1140 = document.getElementById('lista1140');
        const lista1220 = document.getElementById('lista1220');

        lista1140.innerHTML = "";
        lista1220.innerHTML = "";

        // Ordenação alfabética dos inscritos
        inscritos.sort((a, b) => a.nome.localeCompare(b.nome));

        inscritos.forEach(p => {
            const li = document.createElement('li');
            li.textContent = p.nome;
            if (p.horario === "11:40") {
                lista1140.appendChild(li);
            } else {
                lista1220.appendChild(li);
            }
        });
    }

    // Reset automático diário
    function checkReset() {
        const lastReset = localStorage.getItem('lastResetDate');
        const hoje = new Date().toDateString();

        // Se for um novo dia e passar de 12h, limpa a lista para o novo ciclo
        if (lastReset !== hoje && new Date().getHours() >= 12) {
            localStorage.setItem('iern_almoco', "[]");
            localStorage.setItem('lastResetDate', hoje);
        }
    }

    // Inicialização
    popularSelect();
    checkReset();
    atualizarInterface();
    setInterval(atualizarInterface, 30000); // Atualiza o status a cada 30 segundos
</script>
